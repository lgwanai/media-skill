Create a new file `scripts/tts_engines/voice_config.py` with a utility function to load voice configuration from markdown files:

```python
import os
import re
import yaml
from datetime import datetime

def load_voice_config(voice_name: str) -> dict | None:
    """Load voice configuration from markdown file.
    
    Looks for config file at data/voices/{voice_name}/config.md
    
    Args:
        voice_name: Name of the voice to load configuration for.
        
    Returns:
        Dictionary with keys: name, engine, instruct, created, notes
        Returns None if config file doesn't exist.
    """
    config_path = os.path.join(
        os.path.dirname(os.path.dirname(os.path.dirname(__file__))),
        "data", "voices", voice_name, "config.md"
    )
    
    if not os.path.exists(config_path):
        return None
    
    with open(config_path, "r", encoding="utf-8") as f:
        content = f.read()
    
    # Parse YAML frontmatter
    frontmatter = {}
    if content.startswith("---"):
        parts = content.split("---", 2)
        if len(parts) >= 3:
            try:
                frontmatter = yaml.safe_load(parts[1]) or {}
            except:
                pass
    
    # Extract instruct section
    instruct = None
    instruct_match = re.search(r'^## Instruct\s*\n(.+?)(?=^##|\Z)', content, re.MULTILINE | re.DOTALL)
    if instruct_match:
        instruct = instruct_match.group(1).strip()
    
    # Extract notes section
    notes = None
    notes_match = re.search(r'^## Notes\s*\n(.+?)(?=^##|\Z)', content, re.MULTILINE | re.DOTALL)
    if notes_match:
        notes = notes_match.group(1).strip()
    
    return {
        "name": frontmatter.get("name", voice_name),
        "engine": frontmatter.get("engine"),
        "created": frontmatter.get("created"),
        "instruct": instruct,
        "notes": notes,
    }
```

2. Use the instruct field if provided
3. Use the engine field if specified in config

Example integration:

```python
# In the dub command handler
if args.voice:
    from tts_engines import load_voice_config
    
    voice_config = load_voice_config(args.voice)
    if voice_config:
        # Use engine from config if specified
        if voice_config.get("engine") and not args.engine:
            args.engine = voice_config["engine"]
            print(f"Using engine from voice config: {args.engine}")
        
        # Use instruct from config if not provided via CLI
        if voice_config.get("instruct") and not args.instruct:
            args.instruct = voice_config["instruct"]
            print(f"Using instruct from voice config: {args.instruct}")
```

This allows users to:
1. Create a voice config file: `data/voices/my-voice/config.md`
2. Specify default engine and instruct
3. Use with: `python scripts/dubbing.py dub --text "Hello" --voice my-voice`

If `--instruct` is provided on CLI, it overrides the config file.
