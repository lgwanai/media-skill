   ```python
   _, clean_text = EmotionParser.parse_emotion_tags(text)
   ```

2. Resolve ref_audio and ref_text from voice_id:
   ```python
   ref_audio = None
   ref_text = None
   if voice_id.startswith("longcat:"):
       ref_audio = voice_id[8:]  # Strip "longcat:" prefix
       # Look up transcript from meta.json
       voices_dir = self._get_voices_dir()
       for vname in os.listdir(voices_dir):
           vpath = os.path.join(voices_dir, vname)
           meta_path = os.path.join(vpath, "meta.json")
           if os.path.exists(meta_path):
               with open(meta_path, "r", encoding="utf-8") as f:
                   meta = json.load(f)
               if meta.get("local_audio") == ref_audio:
                   ref_text = meta.get("text")
                   break
   ```

3. Load model and tokenize:
   ```python
   self.load_model()
   if self._model is None:
       return False
   
   from transformers import AutoTokenizer
   tokenizer = AutoTokenizer.from_pretrained(self._model.config.text_encoder_model)
   inputs = tokenizer([clean_text], padding="longest", return_tensors="pt")
   inputs = {k: v.to(self._model.device) for k, v in inputs.items()}
   ```

4. Load reference audio (24kHz):
   ```python
   import librosa
   prompt_audio = None
   if ref_audio and os.path.exists(ref_audio):
       audio, _ = librosa.load(ref_audio, sr=24000, mono=True)
       prompt_audio = torch.from_numpy(audio).unsqueeze(0).to(self._model.device)
   ```

5. Run inference with thread-safety:
   ```python
   with self._infer_lock:
       steps = tts_params.get("steps", self.config.get("LONGCAT_STEPS", 16)) if tts_params else 16
       cfg_strength = tts_params.get("cfg_strength", self.config.get("LONGCAT_CFG_STRENGTH", 4.0)) if tts_params else 4.0
       
       output = self._model(
           input_ids=inputs["input_ids"],
           attention_mask=inputs["attention_mask"],
           prompt_audio=prompt_audio,
           steps=steps,
           cfg_strength=cfg_strength,
           guidance_method="apg",  # Use APG for voice cloning
       )
   ```

6. Save output audio:
   ```python
   import soundfile as sf
   waveform = output.waveform.squeeze().detach().cpu().numpy()
   
   # Convert to MP3 if needed
   if output_path.lower().endswith(".mp3"):
       from pydub import AudioSegment
       import tempfile
       with tempfile.NamedTemporaryFile(suffix=".wav", delete=False) as tmp:
           sf.write(tmp.name, waveform, 24000)
           audio = AudioSegment.from_wav(tmp.name)
           audio.export(output_path, format="mp3")
           os.unlink(tmp.name)
   else:
       sf.write(output_path, waveform, 24000)
   ```
